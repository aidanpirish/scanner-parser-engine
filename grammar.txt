<program> ::= <statement> 
            | <statement> <program>

<statement> ::=  <print> ";" 
            | <assignment> ";"

<assignment> ::= name "=" <expr> 

<print> ::= "print" <arguments>

<arguments> ::= <expr> | <string> | 

<string> ::= "\"" chars "\""

<expr> ::= <expr> "+" <term> | <expr> "-" <term>
         | <term> | input

<term> ::= <term> "*" <negation> | <term> "/" <negation>
         | <negation>

<negation> ::= "-" <factor>
         | <factor>

<factor> ::= <paren> "^" <factor>
         | <paren>

<paren> ::= "(" <expr> ")"
         | number
